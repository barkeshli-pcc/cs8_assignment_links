2. Define a Queue class (use upper case Q for the name) as follows: (queue.h)

template 
class Queue
{
public:
    class Iterator{
    public:
        friend class Queue;
        Iterator(){_ptr = NULL;}
        Iterator(node* p){}
        T operator *(){}
        bool is_null(){}
        friend bool operator !=(const Iterator& left,
                                const Iterator& right){}

        friend bool operator ==(const Iterator& left,
                                const Iterator& right){}

        Iterator& operator++(){
        }

        friend Iterator operator++(Iterator& it,
                                   int unused){
        }

    private:
        node* _ptr;
    };

    Queue();

    Queue(const Queue& copyMe);
    ~Queue();
    Queue& operator=(const Queue& RHS);

3. Define a Stack class (upper case S for the name) as follows: (stack.h)

template 
class Stack{
public:
    class Iterator{
    public:
        friend class Stack;
        Iterator(){_ptr = NULL;}
        Iterator(node* p){}
        ITEM_TYPE operator *(){}
        bool is_null(){return _ptr == NULL;}
        friend bool operator !=(const Iterator& left,
                                const Iterator& right){return left._ptr != right._ptr;}

        friend bool operator ==(const Iterator& left,
                                const Iterator& right){}

        Iterator& operator++(){
        }

        friend Iterator operator++(Iterator& it,
                                   int unused){
            assert(it._ptr!=NULL);
        }

    private:
        node* _ptr;
    };

    Stack();
    Stack(const Stack& copyMe);
    ~Stack();
    Stack& operator=(const Stack& RHS);
    ITEM_TYPE top();
    bool empty();
    void push(ITEM_TYPE item);
    ITEM_TYPE pop();
    template
    friend ostream& operator<<(ostream& outs, const Stack& printMe);
    Iterator begin() const;
    Iterator end() const;
    int size() const { return _size; }

private:
    node* _top;
    int _size;
};
